#include <stdio.h>

int main(void)
{
	char name[256];
	printf("이름이 뭐에요?:");
	scanf_s("%s", name, sizeof(name));

	int weight;
	printf("몸무게가 몇이세요?:");
	scanf_s("%d", &weight);

	int age;
	printf("나이가 멏살이에요?:");
	scanf_s("%d", &age);

	float height;
	printf("키가 몇cm 에요?:");
	scanf_s("%f", &height);

	char what[256];
	printf("어떤 일을 하나요?:");
	scanf_s("%s", what, sizeof(what));

	printf("\n\n---정보---\n\n");
	printf("이름: %s\n", name);
	printf("몸무게; %d\n", weight);
	printf("나이: %d\n", age);
	printf("키: %.1f\n", height);
	printf("일: %s\n", what);

	return 0;
}
